        function handleChoiceClick(choice, btnElement) {
            playTone(800, 'triangle', 0.1);
            
            Array.from(els.choiceContainer.children).forEach(child => {
                if(child !== btnElement) child.classList.add('opacity-0', 'pointer-events-none');
            });
            
            setTimeout(() => {
                els.choiceContainer.classList.add('hidden');
                
                gameState.battery = Math.min(100, Math.max(0, gameState.battery + choice.battery));
                gameState.distance = Math.min(100, Math.max(0, gameState.distance + choice.distance));
                gameState.mask = Math.min(100, Math.max(0, gameState.mask + choice.mask));
                
                if (gameState.styleCount[choice.style] !== undefined) {
                    gameState.styleCount[choice.style]++;
                } else {
                    gameState.styleCount['其他']++;

                els.feedbackPanel.classList.add('animate-fade-in');

                loadScenario();
            }
        });

                });
                els.eventChoices.appendChild(btn);
            });
            
            els.randomEventOverlay.classList.remove('hidden');
            els.randomEventOverlay.classList.add('flex');
        }

            els.randomEventOverlay.classList.add('hidden');
            els.randomEventOverlay.classList.remove('flex');

        // --- 工具 ---
        function playTone(freq, type = 'sine', duration = 0.1) {
            try {
                const ctx = new (window.AudioContext || window.webkitAudioContext)();
                const osc = ctx.createOscillator();
                const gain = ctx.createGain();
                osc.type = type;
                osc.frequency.value = freq;
                gain.gain.setValueAtTime(0.05, ctx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + duration);
                osc.connect(gain);
                gain.connect(ctx.destination);
                osc.start();
                osc.stop(ctx.currentTime + duration);
            } catch(e) {}
        }
